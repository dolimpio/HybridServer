CREATE TABLE HTML (
	uuid CHAR(36) PRIMARY KEY,
    content TEXT NOT NULL
);

CREATE TABLE XML (
	uuid CHAR(36) PRIMARY KEY,
    content TEXT NOT NULL
);

CREATE TABLE XSD (
	uuid CHAR(36) PRIMARY KEY,
    content TEXT NOT NULL
);

CREATE TABLE XSLT (
	uuid CHAR(36) PRIMARY KEY,
	xsd CHAR(36),
    content TEXT NOT NULL
);
